            }
        } else {
            if (item.quality < MAX_QUALITY) {
                item.quality = item.quality + 1;
                if (item.name.equals(BACKSTAGE_PASSES.name)) {
                    if (item.sellIn < 11) {
                        if (item.quality < MAX_QUALITY) {
                            item.quality = item.quality + 1;
                        }
                    }
                    
                    if (item.sellIn < 6) {
                        if (item.quality < MAX_QUALITY) {
                            item.quality = item.quality + 1;
                        }
                    }
                }
            }
        }

            if (!item.name.equals(BACKSTAGE_PASSES.name)) {

                    item.quality = item.quality - 1;

                item.quality = 0;
            }
        }
